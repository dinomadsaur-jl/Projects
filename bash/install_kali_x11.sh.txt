#!/data/data/com.termux/files/usr/bin/bash

# =============================================================================
# KALI NETHUNTER – COMPLETE FIXED INSTALLER (NO ROOT ISSUES)
# =============================================================================

# Colors
G='\033[0;32m'; Y='\033[1;33m'; R='\033[0;31m'; B='\033[0;34m'; C='\033[0;36m'; N='\033[0m'

# -----------------------------------------------------------------------------
# SELF-INSTALLATION
# -----------------------------------------------------------------------------
SCRIPT_NAME="kali-x11"
TARGET="$PREFIX/bin/$SCRIPT_NAME"

if [ "$0" != "$TARGET" ]; then
    echo -e "${Y}Installing $SCRIPT_NAME to $TARGET...${N}"
    cp "$0" "$TARGET"
    chmod +x "$TARGET"
    echo -e "${G}✅ Installed. Run '$SCRIPT_NAME' to open menu.${N}"
    echo ""
    exec "$TARGET"
    exit 0
fi

# -----------------------------------------------------------------------------
# FUNCTIONS
# -----------------------------------------------------------------------------

check_kali() { [ -d ~/kali-arm64 ]; }
check_nethunter() { command -v nethunter >/dev/null 2>&1; }

nuclear_cleanup() {
    echo -e "${Y}→ Cleaning processes...${N}"
    pkill -9 -f termux-x11 2>/dev/null
    pkill -9 -f pulseaudio 2>/dev/null
    if check_nethunter; then
        nethunter pkill -9 -f Xorg 2>/dev/null
        nethunter pkill -9 -f xfce4 2>/dev/null
        nethunter pkill -9 -f startxfce4 2>/dev/null
    fi
    rm -f /data/data/com.termux/files/usr/tmp/.X0-lock 2>/dev/null
    echo -e "${G}✅ Done${N}"
    read -p "Press Enter..."
}

install_xfce4() {
    echo -e "${Y}→ Installing XFCE4 in Kali...${N}"
    if check_kali && check_nethunter; then
        # Fix DNS first
        nethunter sudo bash -c "echo 'nameserver 8.8.8.8' > /etc/resolv.conf; echo 'nameserver 8.8.4.4' >> /etc/resolv.conf" 2>/dev/null
        
        # Update and install as normal user (not root)
        nethunter apt update -y
        nethunter apt install -y dbus-x11 xfce4 xfce4-terminal
        
        # Verify installation
        if nethunter which startxfce4 >/dev/null 2>&1; then
            echo -e "${G}✅ XFCE4 installed successfully${N}"
            return 0
        else
            echo -e "${R}❌ XFCE4 installation failed${N}"
            return 1
        fi
    fi
}

fix_xorg_permissions() {
    echo -e "${Y}→ Fixing Xorg permissions...${N}"
    if check_kali && check_nethunter; then
        # Only use root for Xwrapper.config
        nethunter sudo bash -c "echo 'allowed_users=anybody' > /etc/X11/Xwrapper.config; echo 'needs_root_rights=yes' >> /etc/X11/Xwrapper.config; chmod 644 /etc/X11/Xwrapper.config" 2>/dev/null
        
        # Create Xauthority files as normal user
        nethunter bash -c "touch /home/kali/.Xauthority 2>/dev/null; chmod 600 /home/kali/.Xauthority 2>/dev/null" 2>/dev/null
        
        echo -e "${G}✅ Xorg permissions fixed${N}"
    fi
}

# =============================================================================
# FIXED KALI-GUI LAUNCHER (NO ROOT FOR XFCE4)
# =============================================================================
create_kali_gui() {
    echo -e "${Y}→ Creating kali-gui launcher (no root)...${N}"
    
    cat > $PREFIX/bin/kali-gui << 'EOF'
#!/data/data/com.termux/files/usr/bin/sh

G='\033[0;32m'; Y='\033[1;33m'; R='\033[0;31m'; N='\033[0m'

echo "${G}╔════════════════════════════════╗${N}"
echo "${G}║     STARTING KALI DESKTOP      ║${N}"
echo "${G}╚════════════════════════════════╝${N}"
echo ""

# Kill existing processes
pkill -f termux-x11 2>/dev/null
pulseaudio --kill 2>/dev/null

# Kill Kali processes (without sudo)
if command -v nethunter >/dev/null 2>&1; then
    nethunter pkill -f Xorg 2>/dev/null
    nethunter pkill -f xfce4 2>/dev/null
    nethunter pkill -f startxfce4 2>/dev/null
fi
rm -f /data/data/com.termux/files/usr/tmp/.X0-lock 2>/dev/null
sleep 2

# Check if XFCE4 is installed
if command -v nethunter >/dev/null 2>&1; then
    if ! nethunter which startxfce4 >/dev/null 2>&1; then
        echo "${Y}XFCE4 not found. Installing...${N}"
        nethunter apt update
        nethunter apt install -y xfce4 xfce4-terminal dbus-x11
    fi
    
    # Find the exact path to startxfce4
    XFCE4_PATH=$(nethunter which startxfce4 2>/dev/null)
    if [ -z "$XFCE4_PATH" ]; then
        XFCE4_PATH="/usr/bin/startxfce4"
    fi
    echo "${Y}Using XFCE4 at: $XFCE4_PATH${N}"
fi

# Fix Xorg permissions (only need sudo for this one command)
if command -v nethunter >/dev/null 2>&1; then
    nethunter sudo bash -c "echo 'allowed_users=anybody' > /etc/X11/Xwrapper.config 2>/dev/null; echo 'needs_root_rights=yes' >> /etc/X11/Xwrapper.config 2>/dev/null" 2>/dev/null
fi

# Start PulseAudio
echo "${Y}→ Starting PulseAudio...${N}"
pulseaudio --start --load="module-native-protocol-tcp" --exit-idle-time=-1 2>/dev/null

# Start Termux:X11
echo "${Y}→ Starting Termux:X11 server...${N}"
termux-x11 :0 -ac &
sleep 3

# Launch X11 app
echo "${Y}→ Opening Termux:X11 app...${N}"
am start --user 0 -n com.termux.x11/com.termux.x11.MainActivity 2>/dev/null
sleep 2

# Launch desktop (AS NORMAL USER, NOT ROOT)
echo "${Y}→ Launching XFCE4 desktop...${N}"
echo "${G}   Username: kali | Password: kali${N}"
echo "${G}   (wait 10-15 seconds)${N}"
echo ""

# CRITICAL: Run startxfce4 as normal user, not root
if command -v nethunter >/dev/null 2>&1; then
    nethunter bash -c "export DISPLAY=:0; export PULSE_SERVER=127.0.0.1; /usr/bin/startxfce4" &
else
    echo "${R}❌ nethunter command not found${N}"
    exit 1
fi

echo "${G}✅ Kali GUI is starting - check the Termux:X11 app${N}"
echo "${Y}   To stop: kali-stop${N}"
EOF

    chmod +x $PREFIX/bin/kali-gui
    echo -e "${G}✅ kali-gui created${N}"
}

# =============================================================================
# CREATE ADDITIONAL COMMANDS
# =============================================================================
create_additional_commands() {
    # kali-stop
    cat > $PREFIX/bin/kali-stop << 'EOF'
#!/data/data/com.termux/files/usr/bin/sh
echo "Stopping Kali processes..."
pkill -f termux-x11 2>/dev/null
pulseaudio --kill 2>/dev/null
if command -v nethunter >/dev/null 2>&1; then
    nethunter pkill -f Xorg 2>/dev/null
    nethunter pkill -f xfce4 2>/dev/null
fi
rm -f /data/data/com.termux/files/usr/tmp/.X0-lock 2>/dev/null
echo "✅ Kali stopped"
EOF
    chmod +x $PREFIX/bin/kali-stop

    # kali-check
    cat > $PREFIX/bin/kali-check << 'EOF'
#!/data/data/com.termux/files/usr/bin/sh
G='\033[0;32m'; Y='\033[1;33m'; R='\033[0;31m'; N='\033[0m'
echo "${Y}════════ KALI STATUS ════════${N}"
echo ""
[ -d ~/kali-arm64 ] && echo "${G}✅ Kali: Installed${N}" || echo "${R}❌ Kali: Missing${N}"
command -v nethunter >/dev/null && echo "${G}✅ nethunter: OK${N}" || echo "${R}❌ nethunter: Missing${N}"
if command -v nethunter >/dev/null; then
    if nethunter which startxfce4 >/dev/null 2>&1; then
        echo "${G}✅ XFCE4: Installed${N}"
    else
        echo "${R}❌ XFCE4: Missing${N}"
    fi
fi
command -v termux-x11 >/dev/null && echo "${G}✅ termux-x11: OK${N}" || echo "${R}❌ termux-x11: Missing${N}"
[ -d /data/data/com.termux.x11 ] 2>/dev/null && echo "${G}✅ X11 APK: Installed${N}" || echo "${Y}⚠️ X11 APK: Install from GitHub${N}"
echo ""
echo "${Y}Commands: kali-gui, kali-stop, kali-check, kali-fix${N}"
EOF
    chmod +x $PREFIX/bin/kali-check

    # kali-fix
    cat > $PREFIX/bin/kali-fix << 'EOF'
#!/data/data/com.termux/files/usr/bin/sh
echo "Applying fixes..."
pkill -f termux-x11 2>/dev/null
pkill -f pulseaudio 2>/dev/null
if command -v nethunter >/dev/null 2>&1; then
    nethunter pkill -f Xorg 2>/dev/null
    nethunter pkill -f xfce4 2>/dev/null
fi
rm -f /data/data/com.termux/files/usr/tmp/.X0-lock 2>/dev/null

if command -v nethunter >/dev/null 2>&1; then
    nethunter sudo bash -c "echo 'nameserver 8.8.8.8' > /etc/resolv.conf; echo 'nameserver 8.8.4.4' >> /etc/resolv.conf" 2>/dev/null
    nethunter sudo bash -c "echo 'allowed_users=anybody' > /etc/X11/Xwrapper.config; echo 'needs_root_rights=yes' >> /etc/X11/Xwrapper.config" 2>/dev/null
    
    if ! nethunter which startxfce4 >/dev/null 2>&1; then
        echo "Installing XFCE4..."
        nethunter apt update
        nethunter apt install -y xfce4 xfce4-terminal dbus-x11
    fi
fi
echo "✅ Fixes applied. Run: kali-gui"
EOF
    chmod +x $PREFIX/bin/kali-fix

    # Widget shortcut
    mkdir -p ~/.shortcuts
    cp $PREFIX/bin/kali-gui ~/.shortcuts/
    chmod +x ~/.shortcuts/kali-gui
    
    echo -e "${G}✅ Additional commands created${N}"
}

# =============================================================================
# INSTALL KALI
# =============================================================================
install_kali() {
    clear
    echo -e "${Y}════════ INSTALLING KALI ════════${N}"
    echo ""
    
    # Step 1: Termux packages
    echo -e "${Y}[1/5] Installing Termux packages...${N}"
    pkg update -y
    pkg install -y termux-x11-nightly pulseaudio proot-distro wget
    echo -e "${G}✅ Done${N}"
    
    # Step 2: Download installer
    echo -e "${Y}[2/5] Downloading Kali installer...${N}"
    wget -O install-nethunter-termux https://offs.ec/2MceZWr
    chmod +x install-nethunter-termux
    echo -e "${G}✅ Done${N}"
    
    # Step 3: Run installer
    echo -e "${Y}[3/5] Running Kali installer...${N}"
    echo -e "${Y}IMPORTANT: Select option 3 (nano) when prompted${N}"
    echo -e "${C}Press Enter to continue...${N}"
    read -r
    ./install-nethunter-termux
    
    if [ ! -d ~/kali-arm64 ]; then
        echo -e "${R}❌ Installation failed${N}"
        read -p "Press Enter..."
        return 1
    fi
    echo -e "${G}✅ Kali installed${N}"
    
    # Step 4: Install XFCE4
    echo -e "${Y}[4/5] Installing XFCE4 in Kali...${N}"
    install_xfce4
    
    # Step 5: Fix permissions
    echo -e "${Y}[5/5] Fixing permissions...${N}"
    fix_xorg_permissions
    
    # Create commands
    create_kali_gui
    create_additional_commands
    
    echo -e "${G}✅ INSTALLATION COMPLETE${N}"
    echo ""
    echo "Next: Install Termux:X11 APK from GitHub"
    echo "Then run: kali-gui"
    read -p "Press Enter..."
}

# =============================================================================
# APPLY FIXES
# =============================================================================
apply_fixes() {
    clear
    echo -e "${C}════════ APPLYING FIXES ════════${N}"
    echo ""
    if check_kali && check_nethunter; then
        install_xfce4
        fix_xorg_permissions
        create_kali_gui
        echo -e "${G}✅ Fixes applied${N}"
    else
        echo -e "${R}❌ Kali not installed${N}"
    fi
    read -p "Press Enter..."
}

# =============================================================================
# SHOW INSTRUCTIONS
# =============================================================================
show_instructions() {
    clear
    echo "════════════════════════════════"
    echo "📋 INSTRUCTIONS"
    echo "════════════════════════════════"
    echo ""
    echo "1. Install Termux:X11 APK:"
    echo "   https://github.com/termux/termux-x11/releases/tag/nightly"
    echo ""
    echo "2. Commands:"
    echo "   kali-gui   - Start Kali desktop (no root)"
    echo "   kali-stop  - Stop all processes"
    echo "   kali-check - Check status"
    echo "   kali-fix   - Reapply fixes"
    echo ""
    echo "3. Login: kali / kali"
    echo ""
    echo "4. If black screen: Run kali-fix then kali-gui"
    echo ""
    echo "════════════════════════════════"
    read -p "Press Enter..."
}

# =============================================================================
# MAIN MENU
# =============================================================================

while true; do
    clear
    echo "╔════════════════════════════════╗"
    echo "║     KALI NETHUNTER ULTIMATE    ║"
    echo "╠════════════════════════════════╣"
    echo "║ 1) FULL INSTALL                ║"
    echo "║ 2) Create ALL commands         ║"
    echo "║ 3) Remove ALL commands         ║"
    echo "║ 4) Apply ALL Fixes             ║"
    echo "║ 5) Nuclear Cleanup             ║"
    echo "║ 6) Remove EVERYTHING           ║"
    echo "║ 7) Check Status                ║"
    echo "║ 8) Test kali-gui               ║"
    echo "║ 9) Instructions                ║"
    echo "║ 0) Exit                        ║"
    echo "╚════════════════════════════════╝"
    echo ""
    read -p "Choice: " opt

    case $opt in
        1) install_kali ;;
        2) create_kali_gui; create_additional_commands; read -p "Press Enter..." ;;
        3) rm -f $PREFIX/bin/kali-{gui,stop,check,fix} 2>/dev/null; rm -f ~/.shortcuts/kali-gui 2>/dev/null; echo "Removed"; read -p "Press Enter..." ;;
        4) apply_fixes ;;
        5) nuclear_cleanup ;;
        6) rm -rf ~/kali-arm64 2>/dev/null; rm -f ~/install-nethunter-termux 2>/dev/null; rm -f $PREFIX/bin/kali-* 2>/dev/null; echo "Removed everything"; read -p "Press Enter..." ;;
        7) kali-check; read -p "Press Enter..." ;;
        8) kali-gui; read -p "Press Enter..." ;;
        9) show_instructions ;;
        0) exit 0 ;;
        *) echo "Invalid choice"; sleep 1 ;;
    esac
done
