<script>
(function() {
    // DOM elements
    const fileInput = document.getElementById('csvInput');
    const fileStatus = document.getElementById('fileStatus');
    const namePoolContainer = document.getElementById('namePoolContainer');
    const errorMessageDiv = document.getElementById('errorMessage');
    const resetBtn = document.getElementById('resetBtn');
    const spinAndPickBtn = document.getElementById('spinAndPickBtn');
    const winnerDisplay = document.getElementById('winnerDisplay');
    const winnerNameSpan = document.getElementById('winnerNameSpan');
    const spinnerSection = document.getElementById('spinnerSection');
    const runningNameDisplay = document.getElementById('runningNameDisplay');

    // state
    let allOriginalNames = [];        // untouched copy (used for reset reference, but we keep removed in workingList)
    let workingNames = [];            // current available names (gets whittled down)
    let removedNames = new Set();     // track removed winners for UI styling
    let currentFileName = '';
    let spinInterval = null;
    let spinActive = false;

    // helper: clear error
    function setError(text) {
        errorMessageDiv.innerHTML = text ? `<div class="error-message">‚ö†Ô∏è ${text}</div>` : '';
    }

    // stop any ongoing spin
    function stopSpinIfRunning() {
        if (spinInterval) {
            clearInterval(spinInterval);
            spinInterval = null;
        }
        spinActive = false;
        refreshUI();
    }

    // refresh UI: status, name chips with removed style, button, spinner visibility
    function refreshUI() {
        // status line
        if (!currentFileName) {
            fileStatus.innerHTML = `‚è≥ no file loaded`;
        } else {
            const total = allOriginalNames.length;
            const remaining = workingNames.length;
            const removedCount = total - remaining;
            fileStatus.innerHTML = `üìÑ ${currentFileName}  ¬∑  <span class="counter">${remaining} left</span>  (${removedCount} removed)`;
        }

        // name pool chips ‚Äì show allOriginalNames with strikethrough if removed
        if (allOriginalNames.length === 0) {
            namePoolContainer.innerHTML = `<div class="empty-pool-message">‚ú® no names yet ‚Äî load a CSV</div>`;
            spinnerSection.style.display = 'none';
            runningNameDisplay.innerText = '‚èé ready';
        } else {
            const MAX_CHIPS = 80;
            let chipsHtml = '';
            const showNames = allOriginalNames.slice(0, MAX_CHIPS);
            showNames.forEach(name => {
                const safeName = String(name).replace(/[&<>"]/g, function(m) {
                    if (m === '&') return '&amp;';
                    if (m === '<') return '&lt;';
                    if (m === '>') return '&gt;';
                    if (m === '"') return '&quot;';
                    return m;
                });
                const isRemoved = removedNames.has(name);
                const removedClass = isRemoved ? 'removed-chip' : '';
                const badge = isRemoved ? '<span class="removed-badge">out</span>' : '';
                chipsHtml += `<span class="name-chip ${removedClass}">${safeName} ${badge}</span>`;
            });
            if (allOriginalNames.length > MAX_CHIPS) {
                chipsHtml += `<span class="name-chip" style="background:#cfdef5;">+${allOriginalNames.length - MAX_CHIPS} more</span>`;
            }
            namePoolContainer.innerHTML = chipsHtml;
            spinnerSection.style.display = 'block';
        }

        // disable button if workingNames empty OR spin active
        spinAndPickBtn.disabled = (workingNames.length === 0 || spinActive);

        // if no names left, hide winner display maybe, but keep if visible? fine.
        if (workingNames.length === 0) {
            // winner might stay, but we also hide the "spin" area a bit? let's keep spinner but show message
            runningNameDisplay.innerText = '‚úÖ all names removed';
        }
    }

    // reset everything (clear file and internal state)
    function resetState(clearFileInput = true) {
        stopSpinIfRunning();
        if (clearFileInput) {
            fileInput.value = '';
        }
        allOriginalNames = [];
        workingNames = [];
        removedNames.clear();
        currentFileName = '';
        setError('');
        winnerDisplay.style.display = 'none';
        runningNameDisplay.innerText = '‚èé ready';
        refreshUI();
    }

    // CSV parser (robust)
    function parseNamesFromCSV(csvText) {
        const lines = csvText.split(/\r?\n/).filter(line => line.trim() !== '');
        if (lines.length === 0) return [];

        let delimiter = ',';
        if (lines[0].includes(';') && !lines[0].includes(',')) delimiter = ';';
        else if (lines[0].includes('\t')) delimiter = '\t';
        else {
            let commaCnt = 0, semiCnt = 0;
            for (let i=0; i<Math.min(3, lines.length); i++) {
                commaCnt += (lines[i].match(/,/g) || []).length;
                semiCnt += (lines[i].match(/;/g) || []).length;
            }
            if (semiCnt > commaCnt) delimiter = ';';
        }

        const headerParts = lines[0].split(delimiter).map(cell => cell.trim().replace(/^["']|["']$/g, ''));
        const lowerHeader = headerParts.map(h => h.toLowerCase());
        const nameKeywords = ['name','nom','nombre','full name','names','participant','winner'];
        let nameColIndex = 0;
        for (let i=0; i<lowerHeader.length; i++) {
            if (nameKeywords.some(kw => lowerHeader[i].includes(kw))) {
                nameColIndex = i;
                break;
            }
        }

        const hasHeader = lowerHeader.some(h => nameKeywords.some(kw => h.includes(kw))) || headerParts.length > 1;
        const startRow = hasHeader ? 1 : 0;
        const extracted = [];

        for (let i=startRow; i<lines.length; i++) {
            const line = lines[i].trim();
            if (!line) continue;
            const parts = line.split(delimiter).map(v => v.trim().replace(/^["']|["']$/g, ''));
            if (parts.length > nameColIndex && parts[nameColIndex] !== '') {
                extracted.push(parts[nameColIndex]);
            } else if (parts.length === 1 && parts[0] !== '') {
                extracted.push(parts[0]);
            }
        }
        return extracted;
    }

    // file selection
    fileInput.addEventListener('change', function(e) {
        const file = e.target.files[0];
        if (!file) {
            resetState(false);
            return;
        }
        currentFileName = file.name;
        setError('');
        winnerDisplay.style.display = 'none';
        stopSpinIfRunning();
        runningNameDisplay.innerText = '‚èé loading...';

        const reader = new FileReader();
        reader.onload = function(ev) {
            try {
                const names = parseNamesFromCSV(ev.target.result);
                if (names.length === 0) {
                    setError('No valid names found.');
                    allOriginalNames = [];
                    workingNames = [];
                } else {
                    // store unique? but keep as they appear (allow duplicates if needed)
                    allOriginalNames = names.slice();   // original order for display
                    workingNames = names.slice();       // mutable copy
                }
                removedNames.clear();
                runningNameDisplay.innerText = workingNames.length ? 'üé≤ ready to spin' : '‚èé no names';
                refreshUI();
            } catch (err) {
                setError('Parse error: ' + err.message);
                allOriginalNames = [];
                workingNames = [];
                removedNames.clear();
                refreshUI();
            }
        };
        reader.onerror = () => {
            setError('Failed to read file');
            allOriginalNames = [];
            workingNames = [];
            removedNames.clear();
            refreshUI();
        };
        reader.readAsText(file);
    });

    // reset button
    resetBtn.addEventListener('click', function() {
        resetState(true);
    });

    // spin animation & remove winner from list
    function startSpinAndRemove() {
        if (spinActive) return;
        if (workingNames.length === 0) {
            setError('No names remaining');
            return;
        }

        stopSpinIfRunning(); // clean any previous interval
        spinActive = true;
        refreshUI(); // disables button

        winnerDisplay.style.display = 'none';

        const totalRemaining = workingNames.length;

        // Choose final winner index from current workingNames
        const finalWinnerIndex = Math.floor(Math.random() * totalRemaining);
        const finalWinner = workingNames[finalWinnerIndex];

        // Animation parameters
        const MIN_STEPS = 20;
        const MAX_STEPS = 35;
        const spinIntervalTime = 60;
        const totalSpinSteps = Math.floor(Math.random() * (MAX_STEPS - MIN_STEPS + 1)) + MIN_STEPS;
        let currentStep = 0;

        // helper random index from workingNames
        function getRandomName() {
            return workingNames[Math.floor(Math.random() * workingNames.length)];
        }

        spinInterval = setInterval(() => {
            // show random name from current pool
            runningNameDisplay.innerText = getRandomName();
            // blink effect
            runningNameDisplay.classList.add('blink-light');
            setTimeout(() => runningNameDisplay.classList.remove('blink-light'), 40);

            currentStep++;
            if (currentStep >= totalSpinSteps) {
                clearInterval(spinInterval);
                spinInterval = null;

                // REMOVE THE WINNER from working list
                // Filter out the winner (only first occurrence? usually names are unique, but if duplicates exist, remove one instance)
                const indexInWorking = workingNames.indexOf(finalWinner);
                if (indexInWorking !== -1) {
                    workingNames.splice(indexInWorking, 1); // remove one occurrence
                }
                // Add to removed set (for original display strikethrough)
                removedNames.add(finalWinner);

                // Update display
                runningNameDisplay.innerText = finalWinner;
                winnerNameSpan.innerHTML = finalWinner.replace(/[&<>"]/g, function(m) {
                    if (m === '&') return '&amp;'; if (m === '<') return '&lt;'; if (m === '>') return '&gt;'; if (m === '"') return '&quot;'; return m;
                }) + ` <small>(removed ¬∑ ${workingNames.length} left)</small>`;
                winnerDisplay.style.display = 'flex';

                spinActive = false;
                refreshUI();  // updates chips (now with strikethrough), button enabled if names left
            }
        }, spinIntervalTime);
    }

    spinAndPickBtn.addEventListener('click', startSpinAndRemove);

    // initial reset
    resetState(true);
})();
</script>
