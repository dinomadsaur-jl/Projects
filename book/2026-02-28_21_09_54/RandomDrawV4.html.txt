<script>
(function() {
    // DOM
    const fileInput = document.getElementById('csvInput');
    const fileStatus = document.getElementById('fileStatus');
    const namePoolContainer = document.getElementById('namePoolContainer');
    const errorMessageDiv = document.getElementById('errorMessage');
    const resetBtn = document.getElementById('resetBtn');
    const spinAndPickBtn = document.getElementById('spinAndPickBtn');
    const winnerDisplay = document.getElementById('winnerDisplay');
    const winnerNameSpan = document.getElementById('winnerNameSpan');
    const spinnerSection = document.getElementById('spinnerSection');
    const runningNameDisplay = document.getElementById('runningNameDisplay');

    const maxSpeedSlider = document.getElementById('maxSpeedSlider');
    const maxSpeedDisplay = document.getElementById('maxSpeedDisplay');
    const durationSlider = document.getElementById('durationSlider');
    const durationDisplay = document.getElementById('durationDisplay');

    // state
    let allOriginalNames = [];
    let workingNames = [];
    let removedNames = new Set();
    let currentFileName = '';
    let spinInterval = null;
    let spinActive = false;

    // pulse parameters
    let maxInterval = 65;
    let totalSpinDuration = 2000;
    const MIN_INTERVAL = 12;
    const MAX_INTERVAL_LIMIT = 350;

    function updateSliderReadouts() {
        maxInterval = parseInt(maxSpeedSlider.value, 10);
        maxSpeedDisplay.textContent = maxInterval + 'ms';
        totalSpinDuration = parseFloat(durationSlider.value) * 1000;
        durationDisplay.textContent = durationSlider.value + 's';
    }
    maxSpeedSlider.addEventListener('input', updateSliderReadouts);
    durationSlider.addEventListener('input', updateSliderReadouts);
    updateSliderReadouts();

    function setError(text) {
        errorMessageDiv.innerHTML = text ? `<div class="error-message">‚ö†Ô∏è ${text}</div>` : '';
    }

    function stopSpinIfRunning() {
        if (spinInterval) { clearTimeout(spinInterval); spinInterval = null; }
        spinActive = false;
        refreshUI();
    }

    function refreshUI() {
        if (!currentFileName) fileStatus.innerHTML = `‚è≥ no file loaded`;
        else {
            const total = allOriginalNames.length;
            const remaining = workingNames.length;
            fileStatus.innerHTML = `üìÑ ${currentFileName}  ¬∑  <span class="counter">${remaining} left</span>  (${total-remaining} removed)`;
        }

        if (allOriginalNames.length === 0) {
            namePoolContainer.innerHTML = `<div class="empty-pool-message">‚ú® load CSV to see names</div>`;
            spinnerSection.style.display = 'none';
            runningNameDisplay.innerText = 'üé≤ ready';
        } else {
            const MAX_CHIPS = 80;
            let html = '';
            allOriginalNames.slice(0, MAX_CHIPS).forEach(name => {
                const safe = String(name).replace(/[&<>]/g, m => m==='&'?'&amp;':m==='<'?'&lt;':'&gt;');
                const isRemoved = removedNames.has(name);
                const chipClass = isRemoved ? 'name-chip removed-chip' : 'name-chip';
                const badge = isRemoved ? '<span class="removed-badge">out</span>' : '';
                html += `<span class="${chipClass}">${safe} ${badge}</span>`;
            });
            if (allOriginalNames.length > MAX_CHIPS) html += `<span class="name-chip">+${allOriginalNames.length-MAX_CHIPS}</span>`;
            namePoolContainer.innerHTML = html;
            spinnerSection.style.display = 'block';
        }
        spinAndPickBtn.disabled = (workingNames.length === 0 || spinActive);
        if (workingNames.length === 0 && allOriginalNames.length > 0) runningNameDisplay.innerText = '‚úÖ done';
    }

    function resetState(clearFileInput = true) {
        stopSpinIfRunning();
        if (clearFileInput) fileInput.value = '';
        allOriginalNames = []; workingNames = []; removedNames.clear();
        currentFileName = ''; setError(''); winnerDisplay.style.display = 'none';
        runningNameDisplay.innerText = 'üé≤ ready'; refreshUI();
    }

    function parseNamesFromCSV(csvText) {
        const lines = csvText.split(/\r?\n/).filter(l => l.trim() !== '');
        if (!lines.length) return [];
        let delimiter = ',';
        if (lines[0].includes(';') && !lines[0].includes(',')) delimiter = ';';
        else if (lines[0].includes('\t')) delimiter = '\t';
        else {
            const com = lines.slice(0,3).reduce((acc, l) => acc + (l.match(/,/g) || []).length, 0);
            const sem = lines.slice(0,3).reduce((acc, l) => acc + (l.match(/;/g) || []).length, 0);
            if (sem > com) delimiter = ';';
        }
        const headerCols = lines[0].split(delimiter).map(c => c.trim().replace(/^["']|["']$/g, ''));
        const lower = headerCols.map(h => h.toLowerCase());
        const nameKeywords = ['name','nom','nombre','full name','names','participant','winner'];
        let nameIdx = 0;
        for (let i=0; i<lower.length; i++) {
            if (nameKeywords.some(kw => lower[i].includes(kw))) { nameIdx = i; break; }
        }
        const hasHeader = lower.some(h => nameKeywords.some(kw => h.includes(kw))) || headerCols.length > 1;
        const start = hasHeader ? 1 : 0;
        const result = [];
        for (let i=start; i<lines.length; i++) {
            const parts = lines[i].split(delimiter).map(v => v.trim().replace(/^["']|["']$/g, ''));
            if (parts.length > nameIdx && parts[nameIdx] !== '') result.push(parts[nameIdx]);
            else if (parts.length === 1 && parts[0] !== '') result.push(parts[0]);
        }
        return result;
    }

    fileInput.addEventListener('change', function(e) {
        const file = e.target.files[0];
        if (!file) { resetState(false); return; }
        currentFileName = file.name;
        setError(''); winnerDisplay.style.display = 'none'; stopSpinIfRunning();
        runningNameDisplay.innerText = '‚è≥...';
        const reader = new FileReader();
        reader.onload = function(ev) {
            try {
                const names = parseNamesFromCSV(ev.target.result);
                if (!names.length) { setError('No names'); allOriginalNames = []; workingNames = []; }
                else { allOriginalNames = names.slice(); workingNames = names.slice(); }
                removedNames.clear();
                runningNameDisplay.innerText = workingNames.length ? 'üé≤ ready' : '‚èé empty';
                refreshUI();
            } catch { setError('Parse error'); allOriginalNames = []; workingNames = []; removedNames.clear(); refreshUI(); }
        };
        reader.onerror = () => { setError('File error'); resetState(false); };
        reader.readAsText(file);
    });

    resetBtn.addEventListener('click', () => resetState(true));

    // pulse spin
    function startPulseSpin() {
        if (spinActive || !workingNames.length) { setError('No names'); return; }
        stopSpinIfRunning();
        spinActive = true; refreshUI(); winnerDisplay.style.display = 'none';

        const finalWinner = workingNames[Math.floor(Math.random() * workingNames.length)];
        const peak = maxInterval;
        const slowPeak = Math.min(MAX_INTERVAL_LIMIT, peak * 2.8);
        const cycle = 800; // ms per wave
        const duration = totalSpinDuration;
        const start = performance.now();

        function getInterval(elapsed) {
            const t = (elapsed % cycle) / cycle;
            const sin = Math.sin(t * Math.PI); // 0..1..0
            const factor = 0.25 + 0.75 * sin;
            const range = slowPeak - peak;
            return Math.round(peak + range * (1 - factor));
        }

        function step() {
            if (!spinActive) return;
            const now = performance.now();
            const elapsed = now - start;
            if (elapsed >= duration) {
                spinActive = false; spinInterval = null;
                const idx = workingNames.indexOf(finalWinner);
                if (idx !== -1) workingNames.splice(idx, 1);
                removedNames.add(finalWinner);
                runningNameDisplay.innerText = finalWinner;
                winnerNameSpan.innerHTML = finalWinner.replace(/[&<>]/g, m => m==='&'?'&amp;':m==='<'?'&lt;':'&gt;') + ` <span style="font-size:1rem;">‚¨áÔ∏è ${workingNames.length}</span>`;
                winnerDisplay.style.display = 'flex';
                refreshUI();
                return;
            }
            runningNameDisplay.innerText = workingNames[Math.floor(Math.random() * workingNames.length)];
            runningNameDisplay.classList.add('blink-light');
            setTimeout(() => runningNameDisplay.classList.remove('blink-light'), 40);
            const next = Math.max(MIN_INTERVAL, getInterval(elapsed));
            spinInterval = setTimeout(step, next);
        }
        spinInterval = setTimeout(step, 10);
    }

    spinAndPickBtn.addEventListener('click', startPulseSpin);
    resetState(true);
})();
</script>
