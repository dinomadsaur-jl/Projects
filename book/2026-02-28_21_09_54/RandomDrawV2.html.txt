<script>
(function() {
    // DOM elements
    const fileInput = document.getElementById('csvInput');
    const fileStatus = document.getElementById('fileStatus');
    const namePoolContainer = document.getElementById('namePoolContainer');
    const errorMessageDiv = document.getElementById('errorMessage');
    const resetBtn = document.getElementById('resetBtn');
    const spinAndPickBtn = document.getElementById('spinAndPickBtn');
    const winnerDisplay = document.getElementById('winnerDisplay');
    const winnerNameSpan = document.getElementById('winnerNameSpan');
    const spinnerSection = document.getElementById('spinnerSection');
    const runningNameDisplay = document.getElementById('runningNameDisplay');
    const speedSlider = document.getElementById('speedSlider');
    const speedDisplay = document.getElementById('speedDisplay');

    // state
    let allOriginalNames = [];        // keep order for chips
    let workingNames = [];            // mutable (remaining)
    let removedNames = new Set();     // set of removed strings
    let currentFileName = '';
    let spinInterval = null;
    let spinActive = false;

    // speed getter (ms)
    function getCurrentSpeed() {
        return parseInt(speedSlider.value, 10);
    }

    // update speed display
    function updateSpeedLabel() {
        speedDisplay.textContent = getCurrentSpeed() + ' ms';
    }
    speedSlider.addEventListener('input', updateSpeedLabel);
    updateSpeedLabel();

    // helper: stop any ongoing spin
    function stopSpinIfRunning() {
        if (spinInterval) {
            clearInterval(spinInterval);
            spinInterval = null;
        }
        spinActive = false;
        refreshUI();
    }

    // clear error
    function setError(text) {
        errorMessageDiv.innerHTML = text ? `<div class="error-message">‚ö†Ô∏è ${text}</div>` : '';
    }

    // refresh: status, chips, buttons, spinner text
    function refreshUI() {
        // status line
        if (!currentFileName) {
            fileStatus.innerHTML = `‚è≥ no file loaded`;
        } else {
            const total = allOriginalNames.length;
            const remaining = workingNames.length;
            fileStatus.innerHTML = `üìÑ ${currentFileName}  ¬∑  <span class="counter">${remaining} left</span>  (${total-remaining} removed)`;
        }

        // render chips: show allOriginalNames, removed ones with strikethrough
        if (allOriginalNames.length === 0) {
            namePoolContainer.innerHTML = `<div class="empty-pool-message">‚ú® load a CSV to see names</div>`;
            spinnerSection.style.display = 'none';
            runningNameDisplay.innerText = 'üé≤ ready';
        } else {
            const MAX_CHIPS = 90;
            let html = '';
            const sliced = allOriginalNames.slice(0, MAX_CHIPS);
            sliced.forEach(name => {
                const safe = String(name).replace(/[&<>]/g, function(m) {
                    return (m === '&') ? '&amp;' : (m === '<' ? '&lt;' : '&gt;');
                });
                const isRemoved = removedNames.has(name);
                const chipClass = isRemoved ? 'name-chip removed-chip' : 'name-chip';
                const badge = isRemoved ? '<span class="removed-badge">out</span>' : '';
                html += `<span class="${chipClass}">${safe} ${badge}</span>`;
            });
            if (allOriginalNames.length > MAX_CHIPS) {
                html += `<span class="name-chip" style="background:#b9d0ef;">+${allOriginalNames.length - MAX_CHIPS} more</span>`;
            }
            namePoolContainer.innerHTML = html;
            spinnerSection.style.display = 'block';
        }

        // disable spin button if workingNames empty OR spin active
        spinAndPickBtn.disabled = (workingNames.length === 0 || spinActive);

        if (workingNames.length === 0 && allOriginalNames.length > 0) {
            runningNameDisplay.innerText = '‚úÖ all removed';
        }
    }

    // reset everything
    function resetState(clearFileInput = true) {
        stopSpinIfRunning();
        if (clearFileInput) {
            fileInput.value = '';
        }
        allOriginalNames = [];
        workingNames = [];
        removedNames.clear();
        currentFileName = '';
        setError('');
        winnerDisplay.style.display = 'none';
        runningNameDisplay.innerText = 'üé≤ ready';
        refreshUI();
    }

    // CSV parser (smart)
    function parseNamesFromCSV(csvText) {
        const lines = csvText.split(/\r?\n/).filter(l => l.trim() !== '');
        if (lines.length === 0) return [];

        // detect delimiter
        let delimiter = ',';
        if (lines[0].includes(';') && !lines[0].includes(',')) delimiter = ';';
        else if (lines[0].includes('\t')) delimiter = '\t';
        else {
            const com = lines.slice(0,3).reduce((acc, l) => acc + (l.match(/,/g) || []).length, 0);
            const sem = lines.slice(0,3).reduce((acc, l) => acc + (l.match(/;/g) || []).length, 0);
            if (sem > com) delimiter = ';';
        }

        const headerCols = lines[0].split(delimiter).map(c => c.trim().replace(/^["']|["']$/g, ''));
        const lower = headerCols.map(h => h.toLowerCase());
        const nameKeywords = ['name','nom','nombre','full name','names','participant','winner','first'];
        let nameIdx = 0;
        for (let i=0; i<lower.length; i++) {
            if (nameKeywords.some(kw => lower[i].includes(kw))) { nameIdx = i; break; }
        }

        const hasHeader = lower.some(h => nameKeywords.some(kw => h.includes(kw))) || headerCols.length > 1;
        const start = hasHeader ? 1 : 0;
        const result = [];
        for (let i=start; i<lines.length; i++) {
            const parts = lines[i].split(delimiter).map(v => v.trim().replace(/^["']|["']$/g, ''));
            if (parts.length > nameIdx && parts[nameIdx] !== '') {
                result.push(parts[nameIdx]);
            } else if (parts.length === 1 && parts[0] !== '') {
                result.push(parts[0]);
            }
        }
        return result;
    }

    // file handling
    fileInput.addEventListener('change', function(e) {
        const file = e.target.files[0];
        if (!file) {
            resetState(false);
            return;
        }
        currentFileName = file.name;
        setError('');
        winnerDisplay.style.display = 'none';
        stopSpinIfRunning();
        runningNameDisplay.innerText = '‚è≥ reading...';

        const reader = new FileReader();
        reader.onload = function(ev) {
            try {
                const names = parseNamesFromCSV(ev.target.result);
                if (names.length === 0) {
                    setError('No valid names');
                    allOriginalNames = []; workingNames = [];
                } else {
                    allOriginalNames = names.slice();
                    workingNames = names.slice();   // mutable copy
                }
                removedNames.clear();
                runningNameDisplay.innerText = workingNames.length ? 'üé≤ ready' : '‚èé no names';
                refreshUI();
            } catch (err) {
                setError('Parse error: ' + err.message);
                allOriginalNames = []; workingNames = [];
                removedNames.clear();
                refreshUI();
            }
        };
        reader.onerror = () => {
            setError('File read failed');
            allOriginalNames = []; workingNames = [];
            removedNames.clear();
            refreshUI();
        };
        reader.readAsText(file);
    });

    // reset button
    resetBtn.addEventListener('click', function() {
        resetState(true);
    });

    // spin & remove with adjustable speed
    function startSpinAndRemove() {
        if (spinActive) return;
        if (workingNames.length === 0) {
            setError('No names left');
            return;
        }

        stopSpinIfRunning();  // kill any previous
        spinActive = true;
        refreshUI();   // disable button

        winnerDisplay.style.display = 'none';

        const totalRemaining = workingNames.length;
        const finalWinnerIndex = Math.floor(Math.random() * totalRemaining);
        const finalWinner = workingNames[finalWinnerIndex];

        // determine steps (random between 18 and 42)
        const stepsMin = 18, stepsMax = 42;
        const totalSteps = Math.floor(Math.random() * (stepsMax - stepsMin + 1)) + stepsMin;
        let step = 0;

        // helper: random name from working
        const randomName = () => workingNames[Math.floor(Math.random() * workingNames.length)];

        // use current speed from slider each interval (read fresh)
        const spinLoop = () => {
            if (!spinActive) return; // just in case

            // show random name
            runningNameDisplay.innerText = randomName();
            runningNameDisplay.classList.add('blink-light');
            setTimeout(() => runningNameDisplay.classList.remove('blink-light'), 40);

            step++;
            if (step >= totalSteps) {
                clearInterval(spinInterval);
                spinInterval = null;

                // REMOVE winner from working (first occurrence)
                const idx = workingNames.indexOf(finalWinner);
                if (idx !== -1) workingNames.splice(idx, 1);
                removedNames.add(finalWinner);   // mark removed for UI

                // final display
                runningNameDisplay.innerText = finalWinner;
                winnerNameSpan.innerHTML = finalWinner.replace(/[&<>]/g, m => m === '&' ? '&amp;' : m === '<' ? '&lt;' : '&gt;')
                    + ` <span style="font-size:1.2rem; opacity:0.8;">(removed ¬∑ ${workingNames.length} left)</span>`;
                winnerDisplay.style.display = 'flex';

                spinActive = false;
                refreshUI();   // chips get strikethrough, button re-enabled if names left
            } else {
                // continue: set next timeout with current slider speed
                spinInterval = setTimeout(spinLoop, getCurrentSpeed());
            }
        };

        // start first step using current speed
        spinInterval = setTimeout(spinLoop, getCurrentSpeed());
    }

    spinAndPickBtn.addEventListener('click', startSpinAndRemove);

    // initial reset
    resetState(true);
})();
</script>
